func minSubArrayLen(target int, nums []int) int {

    if n == 0 {

    minLen := n + 1
    left, sum := 0, 0
    for right := 0; right < n; right++ {

            sum -= nums[left]
            left++
        }
    }
    if minLen == n+1 {
        return 0
    }
    return minLen
}
